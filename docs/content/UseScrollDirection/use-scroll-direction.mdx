# use-scroll-direction

**Description**

Detects scroll direction

**Props**

- Debounce (ms)

**Returns**

- One of these ('top' | 'up' | 'down')

```js
const scrollingDir = useScrollDirection(debounce);
// returns ('top' | 'up' | 'down') scroll direction
```

```jsx
import React from "react";
import { useScrollDirection } from "@jam3/react-hooks";

const ScrollDirection = () => {
  const scrollingDir = useScrollDirection();

  return <div>Is scrolling (direction): {scrollingDir}</div>;
};
export default ScrollDirection;
```
