# use-on-hover

**Description**

Detects when an element is being hovered, and also takes two callbacks for when is in or out

**Props**

- domRef: DOM Reference
- inCallback: Function to be call when hovering in
- outCallback: Function to be call when hovering out

```js
useOnHover(domRef, inCallback, outCallback);

function inCallback() {
  // do something here when hovering in
}

function outCallback() {
  // do something here when hovering out
}
```

```jsx
import React, { useRef } from "react";
import { useOnHover } from "@jam3/react-hooks";

const YourComponent = () => {
  useOnHover(hoverRef, onHoverIn, onHoverOut);

  function onHoverIn() {
    hoverRef.current.style.backgroundColor = "green";
  }

  function onHoverOut() {
    hoverRef.current.style.backgroundColor = "red";
  }

  return <div ref={hoverRef}>Hover me</div>;
};
```
